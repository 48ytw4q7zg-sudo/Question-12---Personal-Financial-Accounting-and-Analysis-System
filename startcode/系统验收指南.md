```sql
-- 1. 验证新增账户已写入
SELECT id, user_id, name, type, initial_balance, currency, status
FROM account WHERE name = '验收账户';
-- 预期: 返回 1 行，status=1

-- 2. 验证修改成功
SELECT name, initial_balance FROM account WHERE name LIKE '验收账户%';
-- 预期: name='验收账户-改', initial_balance=1500.00

-- 3. 验证软删除 (status=0)
SELECT id, name, status FROM account WHERE name = '验收账户-改';
-- 预期: status=0 (如果已删除)

-- 4. 验证 zhangsan 有 4 个种子账户
SELECT COUNT(*) AS account_count FROM account WHERE user_id = 1 AND status = 1;
-- 预期: account_count = 4 (现金/银行卡/支付宝/微信)

-- 5. 验证账户类型枚举
SELECT type, COUNT(*) FROM account WHERE user_id = 1 GROUP BY type;
-- 预期: 返回 4 行，type 值在 {1,2,3,4} 范围内

-- 6. 验证余额计算: 初始余额 + 收入 - 支出
SELECT
  a.id,
  a.name,
  a.initial_balance,
  COALESCE(SUM(CASE WHEN t.type = 1 THEN t.amount ELSE 0 END), 0) AS total_income,
  COALESCE(SUM(CASE WHEN t.type = 2 THEN t.amount ELSE 0 END), 0) AS total_expense,
  a.initial_balance
    + COALESCE(SUM(CASE WHEN t.type = 1 THEN t.amount ELSE 0 END), 0)
    - COALESCE(SUM(CASE WHEN t.type = 2 THEN t.amount ELSE 0 END), 0) AS current_balance
FROM account a
LEFT JOIN transaction t ON t.account_id = a.id AND t.user_id = a.user_id
WHERE a.user_id = 1 AND a.status = 1
GROUP BY a.id, a.name, a.initial_balance;
-- 预期: current_balance = initial_balance + income - expense
```

#### 4.5 金额精度验收 (DECIMAL(12,2) 存储精度)

> 验证数据库 DECIMAL(12,2) 类型能精确存储和读取两位小数金额，不产生浮点误差。

**测试原理**: 写入 4 个典型精度边界值 (0.01 / 0.10 / 0.99 / 999999.99)，再通过 API 读出，比对写入值与读出值是否完全一致。

**curl 精度测试**:

```bash
# 步骤 1: 依次写入 4 个精度测试值
PRECISION_VALUES=("0.01" "0.10" "0.99" "999999.99")
PRECISION_IDS=()

for val in "${PRECISION_VALUES[@]}"; do
  RESP=$(curl -s -X POST http://localhost:8080/api/v1/transaction \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer $TOKEN" \
    -d "{\"accountId\":1,\"categoryId\":1,\"type\":2,\"amount\":${val},\"note\":\"精度测试-${val}\",\"time\":\"2026-05-23 12:00:00\"}")
  echo "写入 ${val}: $(echo $RESP | grep -o '"amount":[0-9.]*')"
  ID=$(echo $RESP | grep -o '"id":[0-9]*' | grep -o '[0-9]*' | head -1)
  PRECISION_IDS+=($ID)
done

# 步骤 2: 通过 ID 逐一读出并比对
for i in "${!PRECISION_VALUES[@]}"; do
  ID=${PRECISION_IDS[$i]}
  EXPECTED=${PRECISION_VALUES[$i]}
  ACTUAL=$(curl -s "http://localhost:8080/api/v1/transaction/${ID}" \
    -H "Authorization: Bearer $TOKEN" | grep -o '"amount":[0-9.]*' | grep -o '[0-9.]*')
  if [ "$ACTUAL" = "$EXPECTED" ]; then
    echo "✅ ${EXPECTED} → 读出 ${ACTUAL} 一致"
  else
    echo "❌ ${EXPECTED} → 读出 ${ACTUAL} 不一致!"
  fi
done
```

#### 5.5 并发转账死锁验证脚本

> 验证转账服务在并发场景下不会因数据库锁冲突导致死锁。本脚本使用 bash `&` + `wait` 模拟两个用户同时发起方向相反的转账请求。

**前置条件**:
- 账户 A (ID=1, 现金钱包) 余额 >= 300.00
- 账户 B (ID=4, 微信零钱) 余额 >= 300.00
- 如果余额不足，先通过 API 各充值至 >= 300.00

**并发转账死锁测试脚本** (`test_concurrent_transfer.sh`):

```bash
#!/bin/bash
# ============================================================
# 并发转账死锁验证脚本
# 测试场景: 同时发起 A→B 和 B→A 两个方向的转账
# 预期: 不会死锁，所有请求在合理时间内返回 (按 ID 升序加锁)
# ============================================================

BASE_URL="http://localhost:8080/api/v1"

# 获取 token
TOKEN=$(curl -s -X POST "${BASE_URL}/user/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"zhangsan","password":"123456"}' \
  | grep -o '"token":"[^"]*"' | cut -d'"' -f4)

if [ -z "$TOKEN" ]; then
  echo "❌ 登录失败，无法获取 token"
  exit 1
fi
echo "✅ Token 获取成功"

# -------- 场景 1: A→B 和 B→A 同时发起 (经典死锁场景) --------
echo ""
echo "=============================================="
echo "场景 1: A→B 50.00 与 B→A 50.00 同时发起"
echo "=============================================="

START=$(date +%s%N)

# 并发发起两个方向相反的转账请求
curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":1,"toAccountId":4,"amount":50.00,"note":"并发测试:A→B"}' \
  -w "\nA→B: HTTP %{http_code} time=%{time_total}s\n" &

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":4,"toAccountId":1,"amount":50.00,"note":"并发测试:B→A"}' \
  -w "\nB→A: HTTP %{http_code} time=%{time_total}s\n" &

# 等待两个请求都完成 (最长等待 10 秒)
wait

END=$(date +%s%N)
ELAPSED_MS=$(( (END - START) / 1000000 ))
echo ""
echo "⏱ 场景 1 总耗时: ${ELAPSED_MS} ms"

# -------- 场景 2: 同一时刻 A→B 和 A→B (相同方向并发) --------
echo ""
echo "=============================================="
echo "场景 2: A→B 50.00 两笔同时发起 (同方向并发)"
echo "=============================================="

START=$(date +%s%N)

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":1,"toAccountId":4,"amount":50.00,"note":"并发测试:同向1"}' \
  -w "\n同向1: HTTP %{http_code} time=%{time_total}s\n" &

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":1,"toAccountId":4,"amount":50.00,"note":"并发测试:同向2"}' \

wait

END=$(date +%s%N)
ELAPSED_MS=$(( (END - START) / 1000000 ))
echo ""
echo "⏱ 场景 2 总耗时: ${ELAPSED_MS} ms"

# -------- 场景 3: 多账户同时转账 (压力测试) --------
echo ""
echo "=============================================="
echo "场景 3: 4 笔转账同时发起 (压力测试)"
echo "=============================================="

START=$(date +%s%N)

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":1,"toAccountId":2,"amount":10.00,"note":"压力:1→2"}' \
  -w "\n1→2: HTTP %{http_code} time=%{time_total}s\n" &

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":2,"toAccountId":3,"amount":10.00,"note":"压力:2→3"}' \
  -w "\n2→3: HTTP %{http_code} time=%{time_total}s\n" &

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":3,"toAccountId":4,"amount":10.00,"note":"压力:3→4"}' \
  -w "\n3→4: HTTP %{http_code} time=%{time_total}s\n" &

curl -s -X POST "${BASE_URL}/transaction/transfer" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${TOKEN}" \
  -d '{"fromAccountId":4,"toAccountId":1,"amount":10.00,"note":"压力:4→1"}' \

# -------- 结果汇总 --------
echo ""
echo "=============================================="
echo "死锁验证结果判定"
echo "=============================================="
echo "判定标准:"
echo "  ✅ 全部 HTTP 200 → 并发安全，无死锁"
```

### 5.4 CSRF 防护验证

> **本项目使用 JWT Bearer Token 认证，天然防 CSRF 攻击，无需额外 CSRF Token。** 以下解释为什么，以及如何验证。

#### 5.4.1 CSRF 攻击原理回顾

CSRF (Cross-Site Request Forgery) 攻击的典型流程:
1. 用户在 `bank.com` 已登录 (浏览器自动携带 Cookie)
2. 用户访问恶意网站 `evil.com`
3. `evil.com` 上的表单或 JS 自动向 `bank.com/transfer` 发请求
4. 浏览器**自动附带** `bank.com` 的 Cookie → 请求被认证通过 → 攻击成功

CSRF 攻击的**必要条件**是浏览器自动携带认证凭证 (Cookie / HTTP Basic Auth 等)。JWT Bearer Token 不满足此条件。

#### 5.4.2 为什么 JWT Bearer Token 天然防 CSRF

| CSRF 必要条件 | JWT Bearer Token 的情况 | 是否满足 |
|---|---|---|
| 浏览器自动附带凭证 | JWT 存储在 `localStorage`，需**手动**通过 JS 从 localStorage 读取并添加到 `Authorization` Header | **否** |
| 跨站请求能带上凭证 | 即使恶意网站向 `localhost:8080` 发请求，浏览器也不会自动添加 `Authorization: Bearer xxx` Header | **否** |
| Cookie-based Session | 本项目不使用 Cookie，无 Session | **否** |

结论: **CSRF 攻击的三个必要条件本项目都不满足，因此天然免疫 CSRF。** 这是 JWT Bearer Token 方案的架构优势之一。

#### 5.4.3 验证 JWT Token 不会随跨站请求自动发送

```bash
# 验证 1: 无 Authorization Header 的请求被正确拦截
curl -s -X POST http://localhost:8080/api/v1/transaction \
  -H "Content-Type: application/json" \
  -d '{"accountId":1,"categoryId":1,"type":2,"amount":100.00,"note":"CSRF测试","time":"2026-05-23 12:00:00"}'
# 预期: code=401, "未登录或 token 已过期"
# 说明: 不手动添加 Authorization Header 就无法通过认证

# 验证 2: 即使模仿跨站请求 (添加 Origin Header)，也无法绕过 JWT 校验
curl -s -X POST http://localhost:8080/api/v1/transaction \
  -H "Content-Type: application/json" \
  -H "Origin: http://evil.com" \
  -H "Referer: http://evil.com/fake-page" \
  -d '{"accountId":1,"categoryId":1,"type":2,"amount":100.00,"note":"CSRF跨站","time":"2026-05-23 12:00:00"}'
# 预期: code=401
# 说明: 即使伪造 Origin/Referer，只要没有有效 JWT token，请求仍被拦截

# 验证 3: CORS 配置也提供额外防护层 (仅允许白名单 Origin)
curl -s -X OPTIONS http://localhost:8080/api/v1/account \
  -H "Origin: http://evil.com" \
  -H "Access-Control-Request-Method: GET"
# 预期: Access-Control-Allow-Origin 不含 evil.com，或直接返回 403
```

#### 5.4.4 安全边界条件

以下**极端情况**需要额外关注 (本项目不涉及):

| 场景 | 风险 | 本项目的处理 |
|---|---|---|
| Token 存在 Cookie 中 (而非 localStorage) | Cookie 会被浏览器自动发送，存在 CSRF 风险 | 本项目用 `localStorage` 存 token，不用 Cookie |
| 同时使用 Session + JWT | Cookie 中 `JSESSIONID` 可能被 CSRF 利用 | 本项目无 Spring Security 默认 Filter Chain，不创建 Session |
| XSS 漏洞窃取 token | 如果存在 XSS，攻击者可直接读取 localStorage 的 token | 见 5.2.2 XSS 防护验证 |

**验收标准**: ✅ 理解 JWT 防 CSRF 原理 √ 无 Cookie 依赖 √ 无 Authorization Header 请求被 401 拦截 √

```sql
WHERE user_id = 1
ORDER BY time DESC
LIMIT 10;
-- 预期: key 包含 idx_user_time (使用了复合索引)
-- 如果 type=ALL 且 Extra 含 "Using filesort": 全表扫描+文件排序

-- 3. 余额计算: 按 account_id 聚合
EXPLAIN SELECT account_id, SUM(CASE WHEN type=1 THEN amount ELSE 0 END)
FROM transaction WHERE user_id = 1 GROUP BY account_id;
-- 预期: key 使用了某个索引 (idx_user_id 或 idx_account_id)

-- 4. 月度汇总: 按时间范围查询
EXPLAIN SELECT * FROM transaction
WHERE user_id = 1 AND time >= '2026-05-01' AND time < '2026-06-01';
-- 预期: key 包含 idx_user_time

-- 5. 关键词搜索: 备注模糊匹配
EXPLAIN SELECT * FROM transaction
WHERE user_id = 1 AND note LIKE '%外卖%';
-- 预期: key=idx_user_id (使用 user_id 索引缩小范围)
-- 注意: LIKE '%...%' 无法使用索引，但 user_id = 1 能缩小扫描范围

-- 6. 预算唯一约束查询
EXPLAIN SELECT * FROM budget
WHERE user_id = 1 AND category_id = 1 AND month = '2026-05';
-- 预期: key=uk_user_category_month (使用了唯一约束索引)
```

### 8.5 连接池监控 (HikariCP)

> SpringBoot 3 默认使用 HikariCP 连接池。验收时需确认连接池配置合理、无连接泄漏。

#### 8.5.1 HikariCP 配置检查

检查 `application.yml` 或 `application-dev.yml`:

```yaml
spring:
  datasource:
    hikari:
      connection-timeout: 30000      # 等待连接超时 30s
      maximum-pool-size: 10          # 最大连接数 (开发环境 10 足够)
      minimum-idle: 5                # 最小空闲连接
      idle-timeout: 600000           # 空闲超时 10 分钟
      max-lifetime: 1800000          # 连接最大存活 30 分钟
      leak-detection-threshold: 10000 # 泄漏检测 10s (开发环境建议开启)
```

#### 8.5.2 通过 Actuator 检查连接池状态 (如果已启用)

```bash
# 检查 HikariCP 连接池状态

# 或直接查 HikariCP metrics (需引入 micrometer)

```

#### 8.5.3 通过后端日志检查

```bash
# 后端启动时 HikariCP 日志

# HikariPool-1 - Start completed.

# 如果有连接泄漏，日志会输出:

#   说明存在连接泄漏，需检查代码中是否有未关闭的 Connection
```

#### 8.5.4 连接数验证 (SQL)

```sql
-- 查看当前 MySQL 连接数
```

## §11 CI/CD 验收 (可选加分项)

> CI/CD 为课程加分项，非必做。本节提供 GitHub Actions 和 Gitee CI 的配置示例与验收步骤。

### 11.1 GitHub Actions 工作流配置

创建文件 `.github/workflows/ci.yml`:

```yaml
name: Finance System CI

on:
  push:
    branches: [master, main]
  pull_request:
    branches: [master, main]

jobs:
  # 后端构建 + 单元测试
  backend-test:
    runs-on: ubuntu-latest
    services:
      mysql:
        image: mysql:8.4
        env:
          MYSQL_ROOT_PASSWORD: root
          MYSQL_DATABASE: finance_db_test
        ports:
          - 3306:3306
        options: --health-cmd="mysqladmin ping" --health-interval=10s --health-timeout=5s --health-retries=3

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up JDK 21
        uses: actions/setup-java@v4
        with:
          java-version: '21'
          distribution: 'temurin'
          cache: maven

      - name: Init Database
        run: |
          mysql -h127.0.0.1 -uroot -proot finance_db_test < sql/01-init.sql

      - name: Run Backend Tests
        working-directory: backend
        run: mvn test -B
        env:
          SPRING_DATASOURCE_URL: jdbc:mysql://127.0.0.1:3306/finance_db_test
          SPRING_DATASOURCE_USERNAME: root
          SPRING_DATASOURCE_PASSWORD: root

      - name: Upload Test Report
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: backend-test-report
          path: backend/target/surefire-reports/

  # 前端构建
  frontend-build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Node.js 24
        uses: actions/setup-node@v4
        with:
          node-version: '24'
          cache: 'pnpm'
          cache-dependency-path: frontend/pnpm-lock.yaml

      - name: Install pnpm
        run: npm install -g pnpm@10

      - name: Install Dependencies
        working-directory: frontend
        run: pnpm install --frozen-lockfile

      - name: Build
        working-directory: frontend
        run: pnpm build

      - name: Upload Build Artifact
        uses: actions/upload-artifact@v4
        with:
          name: frontend-dist
          path: frontend/dist/
```

## §12 数据库备份恢复与灾备验证

> 财务系统的数据是用户核心资产，数据库备份与恢复能力必须在验收时验证。

### 12.1 备份策略设计

| 备份类型 | 频率 | 保留周期 | 适用场景 |
|---|---|:---:|:---:|---|
| **全量备份** (mysqldump) | 每日 1 次 | 30 天 | 日常数据保护、跨服务器迁移 |
| **增量备份** (binlog) | 实时 (MySQL 自动) | 7 天 | 恢复到任意时间点 (PITR) |
| **手动快照** | 重大变更前 (如批量导入) | 永久保留 | 变更前回滚保险 |

**全量备份命令**:

```bash
# 创建备份目录
mkdir -p backups

# 执行 mysqldump 全量备份 (含存储过程/触发器/事件)
mysqldump -uroot -proot \
  --single-transaction \
  --routines \
  --triggers \
  --events \
  --hex-blob \
  --result-file=backups/finance_db_full_$(date +%Y%m%d_%H%M%S).sql \
  finance_db

# 压缩备份文件 (节省 80%+ 磁盘空间)
gzip backups/finance_db_full_*.sql
```

**Windows PowerShell 备份脚本** (`backups/backup.ps1`):

```powershell
$ErrorActionPreference = "Stop"
$BACKUP_DIR = ".\backups"
$TIMESTAMP = Get-Date -Format "yyyyMMdd_HHmmss"
$FILE = "$BACKUP_DIR\finance_db_full_$TIMESTAMP.sql"

if (-not (Test-Path $BACKUP_DIR)) {
    New-Item -ItemType Directory -Path $BACKUP_DIR | Out-Null
}

Write-Host "正在备份 finance_db ..." -ForegroundColor Green
mysqldump -uroot -proot --single-transaction --routines --triggers `
  --result-file="$FILE" finance_db

if ($LASTEXITCODE -eq 0) {
    $size = (Get-Item $FILE).Length / 1KB
    Write-Host "备份完成: $FILE ($([math]::Round($size, 1)) KB)" -ForegroundColor Green
} else {
    Write-Host "备份失败！请检查 MySQL 服务是否运行" -ForegroundColor Red
}
```
